function getPrimesNumbers(num) {
    const numbers = [], primes = [];
    for (let i = 2; i <= num ; i++) {
        if(!numbers[i]) {
            primes.push(i);
            for (let j = i * i; j <= num; j += i) {
                numbers[j] = true;
            }
        }
    }
    return primes;
}
